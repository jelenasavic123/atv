    <script>
        function showActors(gender) {
            document.getElementById('male-actors').style.display = 'none';
            document.getElementById('female-actors').style.display = 'none';
            
            if (gender === 'male') {
                document.getElementById('male-actors').style.display = 'grid';
            } else if (gender === 'female') {
                document.getElementById('female-actors').style.display = 'grid';
            }
        }

        function toggleSeries(card) {
            const seriesDiv = card.querySelector('.actor-series');
            const isVisible = seriesDiv.style.display === 'block';
            seriesDiv.style.display = isVisible ? 'none' : 'block';
        }
    </script>
